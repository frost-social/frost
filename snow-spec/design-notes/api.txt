component ErrorResponse = {
  [name "エラー内容"]
  error: {
    [name "エラーコード"]
    [description "エラーの内容を識別できる値。"]
    code: number,

    [name "エラーメッセージ"]
    [description "エラーの説明文。"]
    message: string,
  },
};

[name "テスト用"]
GET /echo {
  request {
    headers: {
      [name "認証情報"]
      Authorization: string,
    };

    query: {
      [name "メッセージ"]
      [description "返答するメッセージを設定。"]
      message: string,
    };
  }

  [name "OK"]
  [description "リクエスト成功"]
  response 200 {
    headers: {};

    [format "json"]
    body: {
      message: string,
    };
  }

  [name "Bad Request"]
  [description "無効なリクエスト"]
  response 400 {
    headers: {};

    [format "json"]
    body: ErrorResponse;
  }
}

[name "テスト用"]
POST /echo {
  request {
    headers: {
      [name "認証情報"]
      Authorization: string,
    };

    [format "json"]
    body: {
      [name "メッセージ"]
      [description "返答するメッセージを設定。"]
      message: string,
    };
  }

  [name "OK"]
  [description "リクエスト成功"]
  response 200 {
    headers: {};

    [format "json"]
    body: {
      message: string,
    };
  }

  [name "Bad Request"]
  [description "無効なリクエスト"]
  response 400 {
    headers: {};

    [format "json"]
    body: ErrorResponse;
  }
}
