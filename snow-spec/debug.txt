type ErrorResponse = {
  code: number,
  message: string
};

GET /echo {
  header Authorization: string;

  query {
    field message: string;
  }

  response 200: { };
  response 400: ErrorResponse;
}

POST /echo {
  header Authorization: string;

  body {
    field message: string;
  }

  response 200: { };
  response 400: ErrorResponse;
}
